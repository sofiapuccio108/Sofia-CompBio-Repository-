# Generate two random numbers between 0 and 1 to represent x and y coordinates
coords <- runif(2)
# note: these can be interpreted as (x,y) coordnites 

# r is the distance from the origin to the point, this makes it a polar function if it has a noted direction and angel

# Calculate the radius from the origin (0,0), this way you can have the distance from the origin as well as the angle of direction. 
# This uses the Pythagorean theorem: r = sqrt(x^2 + y^2)
r <- sqrt(coords[1]^2 + coords[2]^2)

# Calculate the angular coordinate (theta) using atan 
theta <- atan(coords[2] / coords[1])
# Combine r and theta into vector representing polar coordinates with both angle and radius 
polar <- c(r = r, theta = theta)

QUESTION 4

# Creating a vector line of Noahs Ark  
# Initial queue (sheep is at the front)
queue <- c("sheep", "fox", "owl", "ant")
# The serpent arrives and gets in line (at the end)
queue <- c(queue, "serpent")
# The sheep (which is the first animal) enters the ark (leaves from the front)
queue <- queue[-1]
# The donkey arrives and gets to the front of the line in front of the rest of the queue? 
queue <- c("donkey", queue)
# The serpent leaves, use exclimation point 
queue <- queue[queue != "serpent"]
# The owl also leaves
queue <- queue[queue != "owl"]
# The aphid arrives and the ant say to cut him in line, use the double == sign to say equal
ant_pos <- which(queue == "ant")
queue <- append(queue, "aphid", after = ant_pos - 1)

# input final queue
queue
[1] "donkey" "fox"    "aphid"  "ant"   
# figure out the position of the ant, aka, what is the position equal to  
which(queue == "aphid")
[1] 3
# he is in position three
